244def getNormalizedDirection(matrix):
245    inVec = Vec2f( -matrix[2], matrix[6])
246    rescale = 1.0 / sqrt( inVec.x()*inVec.x() + inVec.y() * inVec.y())
247    return Vec2f( inVec.x() * rescale, inVec.y() * rescale)
